[20]:

def pairs_to_dm(pairs):
    """
    Create a distance matrix
    """
    rows = [(p.set1_id, p.set2_id, 1-p.jaccard_similarity) for p in pairs]
    df = pd.DataFrame(rows, columns =['subsetA', 'subsetB', 'sim'])
    df = df.pivot("subsetA", "subsetB", "sim")
    return df
