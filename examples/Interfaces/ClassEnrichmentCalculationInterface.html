from oaklib.datamodels.input_specification import InputSpecification
from oaklib import get_adapter
spec = InputSpecification(
    ontology_resources={
        "hp": {
            "selector": "sqlite:obo:hp"
        },
    },
    association_resources={
        "hpoa": {
            "selector": "hpoa:"
        }
    }
)

adapter = get_adapter(spec)
